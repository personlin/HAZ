402  subroutine calc_sum_waacy ( sum, mpdf_param, maxMag, beta, minmag,
403  1 iFlt, iParam, iwidth, faultArea, pRatio)
404 
405  implicit none
406  include 'pfrisk.h'
407 
408  real mpdf_param(max_flt,maxparam,max_width,6),
409  1 beta(max_flt,maxparam,max_width),
410  1 maxmag(max_flt,maxparam,max_width)
411  real minMag(max_flt), Mmin
412  real MaxMagWA, Btail, SigM, Fract_Exp, mChar, b_value, stepM
413  real*8 sum, moment, sum1, sum2
414  real WA_PMag(10000)
415  real cumProb(10000)
416  real mag, M1
417  integer iMag, nMag, iFlt, iParam, iWidth, iMag1
418  real pratio
419  real faultArea, areaRatio, area_rup
420 
421 
422 c Set WAACY model parameters (haz45 version)
423  if ( mpdf_param(iflt,iparam,iwidth,5) .eq. 0. ) then
424  maxmagwa = mpdf_param(iflt,iparam,iwidth,2)
425  else
426  maxmagwa = maxmag(iflt,iparam,iwidth) + mpdf_param(iflt,iparam,iwidth,5)
427  endif
428  if (maxmagwa .gt. mpdf_param(iflt,iparam,iwidth,6) ) then
429  maxmagwa = mpdf_param(iflt,iparam,iwidth,6)
430  endif
431  btail = mpdf_param(iflt,iparam,iwidth,3)
432  sigm = mpdf_param(iflt,iparam,iwidth,1)
433  fract_exp = mpdf_param(iflt,iparam,iwidth,4)
434  mchar = maxmag(iflt,iparam,iwidth)
435  b_value = beta(iflt,iparam,iwidth)/alog(10.0)
436 
437 c start integration at Mag=0 for momment balance
438  mmin = 0
439  stepm =0.001
440  nmag = int( (maxmagwa - mmin) / stepm )
441 c write (*,'( i5)') nMag
442 
443 c Compute the probabilities of magnitudes using WAACY
444 c without correction for rupture past the modelled fault
445  call calc_wa_pmag2 ( mchar, sigm, b_value, btail, fract_exp,
446  1 maxmagwa, mmin, wa_pmag, stepm, nmag )
447 
448 c Adjust the moment release to account for the part of the rupture
449 c that is not modelled.
450 
451 c First, Initialize moment sum
452  sum1 = 0.
453  sum2 = 0.
454  m1 = mchar - 0.25
455 
456 c Compute the moment * mag pdf for the part of the rupture this is modelled.
457 c this is separated into the moment in the exp part (sum1) and the moment
458 c in the char part (sum2)
459  do imag=1,nmag
460  mag = mmin + (imag-0.5)*stepm
461  moment = 10.**(1.5*mag+16.05)
462 
463 c Scale the moment from the eqk for the part that is released on the modelled fault
464 c Just use log(A)= M-4 for now
465  area_rup = 10.**(mag -4)
466  arearatio = area_rup / faultarea
467  if (arearatio .gt. 1. ) then
468  moment = moment / arearatio
469  endif
470  if ( mag .LT. m1 ) then
471  sum1 = sum1 + moment*wa_pmag(imag)
472  else
473  sum2 = sum2 + moment*wa_pmag(imag)
474  endif
475  enddo
476 
477 c Find the fraction of the moment in the exp part after the correction for
478 c moment released in non-modelled ruptures
479  fract_exp = (fract_exp / (sum1 / (sum1 + sum2)) ) * fract_exp
480 
481 c Reset the value in the array so this this corrected fraction is
482 c later used for the mag prob
483  mpdf_param(iflt,iparam,iwidth,4)= fract_exp
484 
485 c Recompute the probabilities of magnitudes using WAACY
486 c with the correction for rupture past the modelled fault
487  call calc_wa_pmag2 ( mchar, sigm, b_value, btail, fract_exp,
488  1 maxmagwa, mmin, wa_pmag, stepm, nmag )
489 
490 c Compute the moment * mag pdf for the part of the rupture this is modelled.
491  sum = 0.
492  do imag=1,nmag
493  mag = mmin + (imag-0.5)*stepm
494  moment = 10.**(1.5*mag+16.05)
495 
496 c Scale the moment from the eqk for the part that is released on the modelled fault
497 c Just use log(A)= M-4 for now
498  area_rup = 10.**(mag -4)
499  arearatio = area_rup / faultarea
500  if (arearatio .gt. 1. ) then
501  moment = moment / arearatio
502  endif
503  sum = sum + moment*wa_pmag(imag)
504  enddo
505 
506 c Find the cumulative rate which will be used to scale the rate from M>0 to M>Mmin
507  cumprob(nmag) = wa_pmag(nmag)
508  do imag=nmag-1,1,-1
509  cumprob(imag) = cumprob(imag+1) + wa_pmag(imag)
510  enddo
511 
512 c Set the ratio of Prob for M>Mmin to M>0
513 c Note: cumProb(1) should be unity.
514  imag1 = int( minmag(iflt) / stepm )
515  pratio = cumprob(imag1) / cumprob(1)
516 
517  return
518  end
