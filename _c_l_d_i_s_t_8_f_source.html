318  dy = (fltgrid_y(icellrupdip,icell) - fltgrid_y(icellrupdip,icell+1))
319  dist = sqrt( dx**2.0 + dy**2.0)
320  c1temp = c1temp + dist
321  enddo
322 C Case where closest point cell is further down strike than hypocenter.
323  else
324  do icell=hascell,icellrupstrike-1, 1
325  dx = (fltgrid_x(icellrupdip,icell) - fltgrid_x(icellrupdip,icell+1))
326  dy = (fltgrid_y(icellrupdip,icell) - fltgrid_y(icellrupdip,icell+1))
327  dist = sqrt( dx**2.0 + dy**2.0)
328  c1temp = c1temp + dist

330 
331  endif
332 
333  return
334  end

374  subroutine calc_lineseg_dist ( x1, y1, z1, x2, y2, z2, x0, y0,
375  1 z0, dist )
376 
377  implicit none
378 
379  real x0, x1, x2, y0, y1, y2, z0, z1, z2, dist
380  real t1, t2, x, y, z, L, L1, L2, d1, d2
381 
382 c Find shortest distance to line (without ends)
383 c Interesection at (x,y,z)
384  if ( z1 .ne. z2 ) then
385  t1 = (x2-x1)/(z2-z1)
386  t2 = (y2-y1)/(z2-z1)
387  z = (z0 - (-z1*t1 + x1 - x0)*t1 - (-z1*t2 + y1 - y0)*t2 )
388  1 / ( t1**2 + t2**2 + 1 )
389  x = t1 * (z-z1) + x1
390  y = t2 * (z-z1) + y1
391  elseif ( y1 .ne. y2 ) then
392  z = z1
393  t1 = (x2-x1)/(y2-y1)
394  y = (y0 - (-y1*t1 + x1 - x0)*t1) / (t1**2 + 1)
395  x = t1 * (y-y1) + x1
396  else
397  z = z1
398  y = y1
399  x = x0
400  endif
401  dist = sqrt( (x-x0)**2 + (y-y0)**2 + (z-z0)**2 )
402 
403 c Check if intersection is outside of edge
404  l = sqrt( (x1-x2)**2 + (y1-y2)**2 + (z1-z2)**2 )
405  l1 = sqrt( (x-x1)**2 + (y-y1)**2 + (z-z1)**2 )
406  l2 = sqrt( (x-x2)**2 + (y-y2)**2 + (z-z2)**2 )
407  if ( l1 .le. l .and. l2 .le. l ) then
408  return
409  endif
410 
411 c Intersection does not hit segment
412  d1 = sqrt( (x0-x1)**2 + (y0-y1)**2 + (z0-z1)**2 )
413  d2 = sqrt( (x0-x2)**2 + (y0-y2)**2 + (z0-z2)**2 )
414  dist = min( d1, d2 )
415 
416  return
417  end

421  subroutine inside_outside ( nSeg, xSeg, ySeg, x0, y0,
422  1 insideFlag )
423 
424  implicit none
425 
426 c declarations passed in
427  integer nSeg
428  real xSeg(1), ySeg(1), x0, y0
429 
430 c declarations passed out
431  integer insideFlag
432 
433 c declarations only used within subroutine
434  integer i
435  real twoPi, pi, theta1, theta2, dTheta, dx1, dx2, dy1, dy2,
436  1 sumtheta, test, tol
437 
438 c This subroutine determines if a point (x0,y0) is inside of
439 c the polygon given by xSeg, ySeg.
440 
441  pi = 3.1415926
442  twopi = 2. * pi
443  sumtheta = 0.
444 
445  do i=1,nseg
446 
447 c Compute Azimuth to ends of segments
448  dy1 = yseg(i) - y0
449  dy2 = yseg(i+1) - y0
450  dx1 = xseg(i) - x0
451  dx2 = xseg(i+1) - x0
452  theta1 = atan2( dy1, dx1 )
453  theta2 = atan2( dy2, dx2 )
454  dtheta = theta2 - theta1
455 
456 c Check if theta range is greater than pi (wrap around)
457  if ( dtheta .gt. pi ) then
458  dtheta = dtheta - twopi
459  elseif ( dtheta .lt. -pi ) then
460  dtheta = dtheta + twopi
461  endif
462 
463 c Compute sum of azimuth ranges
464  sumtheta = sumtheta + dtheta
465  enddo
466 
467 c Determine if point is inside the polygon
468 c If sumTheta = +-2 pi , then point is inside
469  test = abs( abs(sumtheta) - twopi )
470  tol = 0.01
471 
472  if ( test .lt. tol ) then
473  insideflag = 1
474  else
475  insideflag = 0
476  endif
477  return
478  end
