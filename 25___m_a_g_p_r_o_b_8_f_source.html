81  else

83  endif

101  mu = magu

104  zmu = (mu-meanmag)/sigma

106  zmagl=(magl-meanmag)/sigma

109  else

123 c Reset mU to magU if necessary (when last mag is not a full step size)
124  if (mu .gt. magu) then

126  endif

128  deltam2 = mpdf_param(iflt,iparam,iwidth,2)

132 c part of the pdf

134  pmag(iparam,iwidth) =
135  1 exp(-beta1*(ml - minmag(iflt)))
136  1 - exp(-beta1 *(mu - minmag(iflt)))
137  elseif (ml .gt. mag2 ) then
138 c Mag range (mL to mU) is completely in the char mag
139 c part of pdf

142  1 *(mag1-minmag(iflt)))) * magstep(iflt)

145  1 *(mag1-minmag(iflt)))) * (magu-ml)
146  endif
147  else
148 c Mag range (mL to mU) stradles the exponential/char
149 c part of the pdf

151  1 - exp(-beta1 *(mag2 - minmag(iflt)))

153  1 *(mag1-minmag(iflt)))) * (mu-mag2)
154 
155

157 c FOR THE CHARACTERISTIC MAGNITUDE MODEL.
158  pmag(iparam,iwidth) = p1 + p2
159  endif

162  ak = (1-exp(-beta1*(magu-deltam1-minmag(iflt))))
163  1 + beta1*exp(-beta1*(mag1-minmag(iflt))) * deltam1
164  pmag(iparam,iwidth) = pmag(iparam,iwidth) / ak
165 
166 c Set large rup flag for syn rupture
167  if ( mag .gt. mag2) then
168  rup1_flag = 1
169  else
170  rup1_flag = 0
171  endif
172

174 c WG99 model
175  elseif ( magrecur(iflt,iparam,iwidth) .eq. 4. ) then
176 
177 c *** char eqk part - normal distribution
178  meanmag = magu - mpdf_param(iflt,iparam,iwidth,1)
179  sigma = mpdf_param(iflt,iparam,iwidth,2)
180 
181 c Check for Delta Function, i.e., Sigma=0
182  if (sigma .eq. 0.0) then
183  if (meanmag .ge. ml .and. meanmag .lt. mu) then
184 c Set probability of Magnitude equal to 1.0 for this
185 c magnitude bin.
186  pchar = 1.0
187  else
188  pchar = 0.0
189  endif
190  else
191  zml = (ml-meanmag)/sigma
192  zmu = (mu-meanmag)/sigma
193  zmagu=(magu-meanmag)/sigma
194  zmagl=(magl-meanmag)/sigma
195  call s27_ndtr(zml,reaL(pmL),d)
196  call s27_ndtr(zmu,real(pmU),d)
197  call s27_ndtr(zmagl,real(pmagL),d)
198  call s27_ndtr(zmagu,real(pmagU),d)
199

201 c FOR THE MAXIMUM MAGNITUDE MODEL.
202 
203  if (mag .ge. meanmag-mpdf_param(iflt,iparam,iwidth,4)*sigma
204  1 .and. mag .le. meanmag+mpdf_param(iflt,iparam,iwidth,4)*sigma) then
205 c change to truncate only on the high end
206  pchar = (pml - pmu)/(1.-pmagu)
207  else
208  pchar = 0.0
209  endif
210 
211  endif

214 C Reset Maxmim magnitude for Exp part equal to mean mag minus 2.0 sigma
215  magu = meanmag - mpdf_param(iflt,iparam,iwidth,4)*sigma
216 
217  if (beta(iflt,iparam,iwidth) .ne. 0.0) then
218  ak = 1. / (1. - exp(-beta(iflt,iparam,iwidth)
219  1 *(magu - magl)) )
220  else
221  write (*,'( 2x,''Error - b-value = 0.0'')')
222  stop 99
223  endif
224

226 c FOR THE EXPONENTIAL MODEL.
227  if (mag .le. magu) then
228  pexp = ak*(exp(-beta1*(ml - magl))
229  1 - exp(-beta1*(mu - magl)) )
230  else
231  pexp = 0.0
232  endif

235 c that goes into each model
236  if (expmeanmo(iparam,iwidth)+charmeanmo(iparam,iwidth) .ne. 0.0) then
237 
238  c1 = mpdf_param(iflt,iparam,iwidth,5)*charmeanmo(iparam,iwidth) /
239  1 ( (1.0-mpdf_param(iflt,iparam,iwidth,5))*expmeanmo(iparam,iwidth) +
240  2 mpdf_param(iflt,iparam,iwidth,5)*charmeanmo(iparam,iwidth) )
241 
242  pmag(iparam,iwidth) = c1*pexp + (1.0-c1)*pchar
243 
244  else
245  pmag(iparam,iwidth) = 0.0
246  endif
247 
248 c BC Hydro Alternative Characteristic Model
249  elseif ( magrecur(iflt,iparam,iwidth) .eq. 6. ) then
250  betaac = mpdf_param(iflt,iparam,iwidth,2)*alog(10.0)
251  magcut = magu - mpdf_param(iflt,iparam,iwidth,1)
252  if (mag .le. magcut) then
253  if (beta(iflt,iparam,iwidth) .ne. 0.0) then
254  ak = 1. / (1. - exp(-beta(iflt,iparam,iwidth)
255  1 *(magu - magl)) )
256  else
257  write (*,'( 2x,''Error - b-value = 0.0'')')
258  stop 99
259  endif
260 c COMPUTE PROBABILITY OF MAGNITUDE BETWEEN mL AND mU
261 c FOR THE EXPONENTIAL MODEL.
262  pmag(iparam,iwidth) = ak*(exp(-beta1*(ml - magl))
263  1 - exp(-beta1*(mu - magl)) )
264  else
265  if (betaac .ne. 0.0) then
266  ak = 1. / (1. - exp(-betaac*(magu - magcut) ) )
267  else
268  write (*,'( 2x,''Error - b-value = 0.0'')')
269  stop 99
270  endif
271 c COMPUTE PROBABILITY OF MAGNITUDE BETWEEN mL AND mU
272 c FOR THE SECOND EXPONENTIAL MODEL.
273  pmag(iparam,iwidth) = ak*(exp(-betaac*(ml - magcut))
274  1 - exp(-betaac*(mu - magcut)) )
275  endif
276

278  sumnorm = 0.0
279  nsummag = (magu - magl)/magstep(iflt) + 1
280  do imagsum=1,nsummag
281 
282  magnorm = real(magl + magstep(iflt)/2. + (imagsum-1)*magstep(iflt))

284  mlnorm = magnorm - magstep(iflt)/2.
285 
286  betaac = mpdf_param(iflt,iparam,iwidth,2)*alog(10.0)
287  magcut = magu - mpdf_param(iflt,iparam,iwidth,1)
288  if (magnorm .le. magcut) then
289  if (beta(iflt,iparam,iwidth) .ne. 0.0) then
290  aknorm = 1. / (1. - exp(-beta(iflt,iparam,iwidth)
291  1 *(magu - magl)) )
292  else
293  write (*,'( 2x,''Error - b-value = 0.0'')')
294  stop 99
295  endif
296 c COMPUTE PROBABILITY OF MAGNITUDE BETWEEN mL AND mU
297 c FOR THE EXPONENTIAL MODEL.
298  pmagnorm(iparam,iwidth) = aknorm*(exp(-beta1*(mlnorm - magl))
299  1 - exp(-beta1*(munorm - magl)) )
300  else
301  if (betaac .ne. 0.0) then
302  aknorm = 1. / (1. - exp(-betaac*(magu - magcut) ) )
303  else
304  write (*,'( 2x,''Error - b-value = 0.0'')')
305  stop 99
306  endif
307 c COMPUTE PROBABILITY OF MAGNITUDE BETWEEN mL AND mU
308 c FOR THE SECOND EXPONENTIAL MODEL.
309  pmagnorm(iparam,iwidth) = aknorm*(exp(-betaac*(mlnorm - magcut))
310  1 - exp(-betaac*(munorm - magcut)) )
311  endif
312 
313  sumnorm = sumnorm + pmagnorm(iparam,iwidth)
314 
315  enddo
316 
317 C Now normalize BC Hydro Model by total
318  pmag(iparam,iwidth) = pmag(iparam,iwidth)/sumnorm
319 
320 c WAACY Model
321  elseif ( magrecur(iflt,iparam,iwidth) .eq. 10. ) then
322  if ( mpdf_param(iflt,iparam,iwidth,5) .eq. 0. ) then
323  maxmagwa = mpdf_param(iflt,iparam,iwidth,2)
324  else
325  maxmagwa = maxmag(iflt,iparam,iwidth) + mpdf_param(iflt,iparam,iwidth,5)
326  endif
327  if (maxmagwa .gt. mpdf_param(iflt,iparam,iwidth,6) ) then
328  maxmagwa = mpdf_param(iflt,iparam,iwidth,6)
329  endif
330  btail = mpdf_param(iflt,iparam,iwidth,3)
331  sigm = mpdf_param(iflt,iparam,iwidth,1)
332  fract_exp = mpdf_param(iflt,iparam,iwidth,4)
333  mchar = maxmag(iflt,iparam,iwidth)
334  b_value = beta(iflt,iparam,iwidth)/alog(10.0)
335  if ( maxmagwa .lt. mchar ) then
336  write (*,'( 2x,''bad parameter for WAACY model, maxmag is too small'')')
337  stop 99
338  endif
339 
340 c Brute force calc of Pmag. Fix this later to be faster.
341  call s25_calc_wa_pmag1 ( mchar, sigm, b_value, btail, fract_exp, maxmagwa,
342  1 minmag(iflt), ml, mu, pmag1)
343  pmag(iparam,iwidth) = pmag1
344 
345  endif
346  endif
347 
348  enddo
349  enddo
350 
351  endif
352 
353  100 continue
354 
355  return
356  end

360  subroutine s25_calc_wa_pmag1 ( mChar, sigM, b_value, bTail, F, Mmax,
361  1 Mmin, ML, MU, WA_Pmag)
362 
363  implicit none
364  real mChar, sigM, b_value, btail, Mmax, F
365  real M1, M2
366  real Mmin, ML, MU, WA_Pmag
367  real pdf1, PHI_15, PHI1, x, step1, mag1
368  real beta, betaTail, c1, c2, c3, d1, d2, alpha, t1
369  real Mo_bar_exp1, Mo_bar_exp2, Mo_bar_char
370  integer nStep, i
371  real temp1, temp2, oneMinusAlpha, x2, x3, x4
372 
373  betatail = btail * alog(10.)
374  beta = b_value * alog(10.)
375 
376 c Calculate WAACY terms
377  m1 = mchar - 0.25
378  m2 = mchar + 1.5*sigm
379  t1 = sqrt(2*3.1415926)
380  c1 = (1./ (t1*sigm)) * exp( -((1.5*sigm)**2) / (2*sigm**2) )
381  x = -0.25 / sigm
382  phi1 = 0.159
383 c call S27_NDTR(x,PHI1)
384  phi_15 = 0.933
385  c2 = 1./ ( 1.-phi1-(1.-phi_15) )
386  c3 = (1. - exp(-betatail*(mmax-m2)) ) / betatail
387 
388  temp1 = beta * 10.**(16.05) * ( exp((-beta+3.45)*m1)-1. )
389  temp2 = ( 1.-exp(-beta*m1))*(-beta+3.45)
390  mo_bar_exp1 = temp1/temp2
391 
392  mo_bar_exp2 = betatail * 10**16.05 * exp( betatail*m2 ) *
393  1 ( exp(( -betatail+3.45)*mmax) - exp(( -betatail+3.45)*m2) ) /
394  2 ( ( 1. - exp(-betatail*(mmax-m2)) ) * ( -betatail + 3.45 ) )
395  mo_bar_char = 10.**(1.5*mchar+16.05) * (2.63*(sigm-0.2) + 1.19)
396 
397  d1 = mo_bar_exp1 / ( exp( -beta*mmin) - exp(-beta*m1) )
398  d2 = (1./ ( 1. + c1*c2*c3)) * mo_bar_char + c1*c2*c3*mo_bar_exp2 / ( 1. + c1*c2*c3 )
399  alpha = d2*f / ( d1*(1.-f)+d2*f )
400  oneminusalpha = d1*(1.-f) / ( d1*(1.-f)+d2*f )
401 
402 c Compute the pdf with a small step size
403  step1 = 0.001
404  nstep = int( (mu - ml ) / step1 )
405 
406 c Compute the probability of mag bewteen ML and MU
407  wa_pmag = 0.
408  do i=1,nstep
409  mag1 = ml + (i-0.5)*step1
410  if ( mag1 .lt. mmin) then
411  pdf1 = 0.
412  elseif ( mag1 .lt. m1 ) then
413  pdf1 = alpha * beta*exp(-beta*(mag1-mmin)) / ( 1. - exp(-beta*(m1-mmin)) )
414  elseif ( mag1 .lt. m2 ) then
415  x2 = c2 / (1. + c1*c2*c3)
416  x3 = 1./ (t1*sigm)
417  x4 = exp( -((mag1-mchar)**2) / (2*sigm**2) )
418  pdf1 = oneminusalpha * x2 * x3 * x4
419 
420  elseif ( mag1 .lt. mmax ) then
421  pdf1 = oneminusalpha * c1*c2 / ( (1.+c1*c2*c3) ) * exp( -betatail*(mag1-m2) )
422  else
423  pdf1 = 0.
424  endif
425  wa_pmag = wa_pmag + pdf1*step1
426  enddo
427 
428  return
429  end

433  subroutine s25_calc_wa_pmag2 ( mChar, sigM, b_value, bTail, F, Mmax,
434  1 Mmin, WA_Pmag, stepM, nMag)
435 
436  implicit none
437  real mChar, sigM, b_value, btail, Mmax, F
438  real M1, M2, d
439  real Mmin, stepM, WA_Pmag(1)
440  real pdf1(10000), PHI_15, PHI1, x, mag1
441  real*8 beta, betaTail, c1, c2, c3, t1
442  real*8 d1, d2, alpha, a1, a2, alpha1, sum1, sum2
443  real*8 Mo_bar_exp1, Mo_bar_exp2, Mo_bar_char
444  integer i, nMag
445  real*8 temp1, temp2
446  betatail = btail * alog(10.)
447  beta = b_value * alog(10.)
448 
449 c Calculate WAACY terms
450  m1 = mchar - 0.25
451  m2 = mchar + 1.5*sigm
452  t1 = sqrt(2*3.1415926)
453  c1 = (1./ (t1*sigm)) * exp( -((1.5*sigm)**2) / (2*sigm**2) )
454  x = -0.25 / sigm
455  call s27_ndtr(x,phi1,d)
456  phi1 = 1-phi1
457  phi_15 = 0.933
458  c2 = 1./ ( phi_15-phi1)
459  c3 = (1. - exp(-betatail*(mmax-m2)) ) / betatail
460 
461 c Mo_bar_exp1 = beta * 10.**(16.05) * ( exp((-beta+3.45)*M1)-1. ) /
462 c 1 ( 1.-exp(-beta*M1)) * (-beta+3.45)
463 
464  temp1 = beta * 10.**(16.05) * ( exp((-beta+3.45)*m1)-1. )
465  temp2 = ( 1.-exp(-beta*m1))*(-beta+3.45)
466  mo_bar_exp1 = temp1/temp2
467 
468  mo_bar_exp2 = betatail * 10**(16.05) * exp( betatail*m2 ) *
469  1 ( exp(( -betatail+3.45)*mmax) - exp(( -betatail+3.45)*m2) ) /
470  2 ( ( 1. - exp(-betatail*(mmax-m2)) ) * ( -betatail + 3.45 ) )
471  mo_bar_char = 10.**(1.5*mchar+16.05) * (2.63*(sigm-0.2) + 1.19)
472 
473  d1 = mo_bar_exp1 / ( exp( -beta*mmin) - exp(-beta*m1) )
474  temp1 = c1*c2*c3
475  d2 = (1./ ( 1. + c1*c2*c3)) * mo_bar_char + c1*c2*c3*mo_bar_exp2 / ( 1. + c1*c2*c3 )
476  a1 = d2*f
477  a2 = d1*(1.-f)+d2*f
478  alpha = d2*f / ( d1*(1.-f)+d2*f )
479  alpha1 = d1*(1-f) / (d1*(1.-f)+d2*f)
480 
481 c Compute the pdf with a small step size
482  do i=1,nmag
483  mag1 = mmin + (i-1)*stepm
484  if ( mag1 .lt. mmin) then
485  pdf1(i) = 0.
486  elseif ( mag1 .lt. m1 ) then
487  pdf1(i) = alpha * beta*exp(-beta*(mag1-mmin)) / ( 1. - exp(-beta*(m1-mmin)) )
488 
489  elseif ( mag1 .lt. m2 ) then
490  pdf1(i) = alpha1 * c2 / ( (1.+c1*c2*c3) * t1*sigm ) *
491  1 exp( -((mag1-mchar)**2) / (2*sigm**2) )
492 
493  elseif ( mag1 .lt. mmax ) then
494  pdf1(i) = alpha1 * c1*c2 / (1.+c1*c2*c3) * exp( -betatail*(mag1-m2) )
495 
496  else
497  pdf1(i) = 0.
498  endif
499 
500  enddo
501 
502 c Compute the probability of mag
503 c Also compute the moment less than M1 (sum1)
504 c and momment greater than M1 (sum2)
505  sum1 = 0.
506  sum2 = 0.
507  do i=1,nmag
508  mag1 = mmin + (i-1)*stepm
509  wa_pmag(i) = pdf1(i)*stepm
510  if (mag1 .le. m1 ) then
511  sum1 = sum1 + 10.**(1.5*mag1+16.05)*wa_pmag(i)
512  else
513  sum2 = sum2 + 10.**(1.5*mag1+16.05)*wa_pmag(i)
514  endif
515  enddo
516 
517  return
518  end
