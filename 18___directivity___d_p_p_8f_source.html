81 c add 0.5 km for the grid spacing to correct for use of centers
82  e = e + 0.5

94  1 - zs * alog( (l2 + rhypo) / (l2 - e + rd) ) )

98 
99 c Equation 6.9

101  iphi = 0
102  else
103  iphi = sinphi * ( zs * alog( (l2 + rhypo) / (l2 - e + rd) ) )
104  endif
105 
106 c Equation 6.11
107  fs_bar = sqrt( ix**2 + in**2 + iphi**2 ) / e
108

117  ctildap = 1.0/ctildap
118  endif

121 
122 C Compute DPP prime (Eq. 6.12).
123  dpp = alog( ctildap * max(e,0.1*ruplength) * max(fs_bar, 0.2) )
124
