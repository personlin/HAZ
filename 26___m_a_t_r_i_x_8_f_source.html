25  subroutine s26_simul( N, A, X, EPS, INDIC, NRC , DETER )
26 C
27 C WHEN INDIC IS NEGATIVE, SIMUL COMPUTES THE INVERSE OF THE N BY
28 C N MATRIX A IN PLACE. WHEN INDIC IS ZERO, SIMUL COMPUTES THE
29 C N SOLUTIONS X(1)...X(N) CORRESPONDING TO THE SET OF LINEAR
30 C EQUATIONS WITH AUGMENTED MATRIX OF COEFFICIENTS IN THE N BY
31 C N+1 ARRAY A AND IN ADDITION COMPUTES THE INVERSE OF THE
32 C COEFFICIENT MATRIX IN PLACE AS ABOVE. IF INDIC IS POSITIVE,
33 C THE SET OF LINEAR EQUATIONS IS SOLVED BUT THE INVERSE IS NOT
34 C COMPUTED IN PLACE. THE GAUSS-JORDAN COMPLETE ELIMINATION METHOD
35 C IS EMPLOYED WITH THE MAXIMUM PIVOT STRATEGY. ROW AND COLUMN
36 C SUBSCRIPTS OF SUCCESSIVE PIVOT ELEMENTS ARE SAVED IN ORDER IN
37 C THE IROW AND JCOL ARRAYS RESPECTIVELY. K IS THE PIVOT COUNTER,
38 C PIVOT THE ALGEBRAIC VALUE OF THE PIVOT ELEMENT, MAX
39 C THE NUMBER OF COLUMNS IN A AND DETER THE DETERMINANT OF THE
40 C COEFFICIENTS MATRIX. THE SOLUTIONS ARE COMPUTED IN THE (N+1)TH
41 C COLUMN OF A AND THEN UNSCRAMBLED AND PUT IN PROPER ORDER IN
42 C X(1)...X(N) USING THE PIVOT SUBSCRIPT INFORMATION AVAILABLE
43 C IN THE IROW AND JCOL ARRAYS. THE SIGN OF THE DETERMINANT IS
44 C ADJUSTED, IF NECESSARY, BY DETERMINING IF AN ODD OR EVEN NUMBER
45 C OF PAIRWISE INTERCHANGES IS REQUIRED TO PUT THE ELEMENTS OF THE
46 C JORD ARRAY IN ASCENDING SEQUENCE WHERE JORD(IROW(I)) = JCOL(I).
47 C IF THE INVERSE IS REQUIRED, IT IS UNSCRAMBLED IN PLACE USING
48 C Y(1)...Y(N) AS TEMPORARY STORAGE. THE VALUE OF THE DETERMINANT
49 C IS RETURNED AS THE VALUE OF THE FUNCTION. SHOULD THE POTENTIAL
50 C PIVOT OF LARGEST MAGNITUDE BE SMALLER IN MAGNITUDE THAN EPS,
51 C THE MATRIX IS CONSIDERED TO BE SINGULAR AND A TRUE ZERO IS
52 C RETURNED AS THE VALUE OF THE FUNCTION.
53 C
54  IMPLICIT REAL*8(a-h, o-z)
55  real*8 A(nrc,nrc), X(n), Y(200), eps, deter
56  integer IROW(200), JCOL(200), JORD(200)
57 C
58  max = n
59  IF ( indic.GE.0 ) max = n + 1
60 C
61 C ..... IS N LARGER THAN 78 .....
62  IF ( n.LE.78 ) GO TO 5
63  WRITE (6,200)
64  deter = 0.
65  RETURN
66 C
67 C ..... BEGIN ELIMINATION PROCEDURE .....
68  5 deter = 1.
69  DO 18 k = 1, n
70  km1 = k - 1
71 C ..... SEARCH FOR THE PIVOT ELEMENT .....
72  pivot = 0.
73  DO 11 i = 1, n
74  DO 11 j = 1, n
75 C ..... SCAN IROW AND JCOL ARRAYS FOR INVALID PIVOT SUBSCRIPTS ....
76  IF ( k.EQ.1 ) GO TO 9
77  DO 8 iscan = 1, km1
78  DO 8 jscan = 1, km1
79  IF ( i.EQ.irow(iscan) ) GO TO 11
80  IF ( j.EQ.jcol(jscan) ) GO TO 11
81  8 CONTINUE
82  9 CONTINUE
83  pivot = a(i,j)
84  irow(k) = i
85  jcol(k) = j
86  11 CONTINUE
87 C
88 C ..... INSURE THAT SELECTED PIVOT IS LARGER THAN EPS .....
89  IF ( dabs(pivot).GT.eps ) GO TO 13
90  deter = 0.
91  RETURN
92 C
93 C ..... UPDATE THE DETERMINANT VALUE .....
94  13 irowk = irow(k)
95  jcolk = jcol(k)
96  deter = deter*pivot
97 C
98 C ..... NORMALIZE PIVOT ROW ELEMENTS .....
99  DO 14 j = 1, max
100  14 a(irowk,j) = a(irowk,j)/pivot
101 C
102 C ..... CARRY OUT ELIMINATION AND DEVELOP INVERSE .....
103  a(irowk,jcolk) = 1./pivot
104  DO 18 i = 1, n
105  aijck = a(i,jcolk)
106  IF ( i.EQ.irowk ) GO TO 18
107  a(i,jcolk) = - aijck/pivot
108  DO 17 j = 1, max
109  17 IF ( j.NE.jcolk ) a(i,j) = a(i,j) - aijck*a(irowk,j)
110  18 CONTINUE
111 C
112 C ..... ORDER SOLUTION VALUES (IF ANY) AND CREATE JORD ARRAY .....
113  DO 20 i = 1, n
114  irowi = irow(i)
115  jcoli = jcol(i)
116  jord(irowi) = jcoli
117  20 IF ( indic.GE.0 ) x(jcoli) = a(irowi,max)
118 C
119 C ..... ADJUST SIGN OF DETERMINANT .....
120  intch = 0
121  nm1 = n - 1
122  DO 22 i = 1, nm1
123  ip1 = i + 1
124  DO 22 j = ip1,n
125  IF ( jord(j).GE.jord(i) ) GO TO 22
126  jtemp = jord(j)
127  jord(j) = jord(i)
128  jord(i) = jtemp
129  intch = intch + 1
130  22 CONTINUE
131  IF( intch/2*2.NE.intch ) deter = - deter
132 C
133 C ..... IF INDIC IS POSITIVE RETURN WITH RESULTS .....
134  IF ( indic.LE.0 ) GO TO 26
135 c DETER = DETER
136  RETURN
137 C
138 C ..... IF INDIC IS NEGATIVE OR ZERO, UNSCRAMBLE THE INVERSE
139 C FIRST BY ROWS .....
140  26 DO 28 j = 1, n
141  DO 27 i = 1, n
142  irowi = irow(i)
143  jcoli = jcol(i)
144  27 y(jcoli) = a(irowi,j)
145  DO 28 i = 1, n
146  28 a(i,j) = y(i)
147 C ..... THEN BY COLUMNS .....
148  DO 30 i = 1, n
149  DO 29 j = 1, n
150  irowj = irow(j)
151  jcolj = jcol(j)
152  29 y(irowj) = a(i,jcolj)
153  DO 30 j = 1, n
154  30 a(i,j) = y(j)
155 C
156 C ..... RETURN FOR INDIC NEGATIVE OR ZERO .....
157 c DETER = DETER
158  RETURN
159 C
160 C ..... FORMAT FOR OUTPUT STATEMENT .....
161  200 FORMAT( 10h0n too big )
162 C
163  END
